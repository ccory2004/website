      print("Fibonacci Sequence")

      #getting correct input from the user
      while True:
          try:
              num = int(input("Enter the number of Fibonacci Seqence numbers to print out: "))
              if num < 2:
                  print("Please enter an integer that is 2 or more")
              else:
                  break
          except:
              print("Please enter a valid input in the form of an integer")
              pass

      #actually calculating the numbers based on the number of numbers requested... thats a lot of numbers
      fibNum = [0, 1]
      print("")
      print("Fibonacci Sequence: ")
      for i in range(num):
          if i != num-1:
              print(str(fibNum[0]) + ", ", end ="")
          else: #handling the last number printed, no comma there, and print new line so the prompt is on the next line
              print(str(fibNum[0]))

          #swtiching the numbers in the array for the next two numbers
          temp = fibNum[0] + fibNum[1]
          fibNum[0] = fibNum[1]
          fibNum[1] = temp
      print("")
